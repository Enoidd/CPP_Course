/*

Una classe è un modello per creare oggetti (Simile alla struct) e contiene:alignas

- Dati (chiamati attributi o membri dati)
- Funzioni (chiamate metodi)

SINTASSI BASE:

Class NomeClasse{
    public:
        // Attributi
        int eta;
        string nome;

        // Metodi
        void saluta(){
        cout << "Ciao, sono "<< nome << "!";
        }
};

CREARE UN OGGETTO DI UNA CLASSE

NomeClasse oggetto;
oggetto.eta = 30;
oggetto.nome = "Luca";
oggetto.saluta(); // fa riferimento al metodo della classe

REGOLE FONDAMENTALI

I membri 'pubblici' (public) possono essere letti o modificati dall'esterno
I membri 'privati' (private) non possono essere accessibili direttamente da fuori (vedi incapsulamento)

==================================
ESEMPIO COMPLETO CON METODI
==================================

#include <iostream>

using namespace std;

class Studente{
    public:

    string nome;
    int eta;
    float media;

    void saluta(){
        cout << "Ciao, sono " << nome << "!";
    }

    void info(){
        cout << "Nome: " << nome << "\nEtà: " << eta << "\nMedia: " << media;
    }
};

int main(){
    Studente s1;
    s1.nome = "Luca";
    s1.eta = 22;
    s1.media = 28.5;

    s1.saluta();
    s1.info();
    return 0;
}

DOVE VENGONO DICHIARATI I METODI?

    - Possono essere dichiarati fuori dalla classe:

    class Prova{
    };

    void Prova::saluta(){
        cout << "Ciao!" << endl;
    }

    - Possono essere dichiarati nella classe:

    class Prova{
        public:
            void saluta(); // solo dichiarazione
            
            
    };

Gli oggetti vengono allocati sullo stack se dichiarati normalmente, sul heap se creati con "new"
